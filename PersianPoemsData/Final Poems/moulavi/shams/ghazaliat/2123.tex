\begin{center}
\section*{غزل شماره ۲۱۲۳: العشق یقول لی تزین}
\label{sec:2123}
\addcontentsline{toc}{section}{\nameref{sec:2123}}
\begin{longtable}{l p{0.5cm} r}
العشق یقول لی تزین
&&
الزینه عندنا تیقن
\\
لا تنظر غیرنا فتعمی
&&
لا تله عن الیقین بالظن
\\
لا عیش لخایف کئیب
&&
لا تبرح عندنا فتامن
\\
من کنت هواه کیف یهلک
&&
من کنت مناه کیف یحزن
\\
العقل رسولنا الیکم
&&
ذاک حسن و نحن احسن
\\
اخشوشن بالبلا و ارضی
&&
فالهجر من البلاء اخشن
\\
من رام الی العلی عروجا
&&
هذا سبب الیه یرکن
\\
یا مضطربا تعال و افلح
&&
فی مسکننا و نعم مسکن
\\
\end{longtable}
\end{center}
