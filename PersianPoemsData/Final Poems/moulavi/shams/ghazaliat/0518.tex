\begin{center}
\section*{غزل شماره ۵۱۸: نفسی بهوی الحبیب فارت}
\label{sec:0518}
\addcontentsline{toc}{section}{\nameref{sec:0518}}
\begin{longtable}{l p{0.5cm} r}
نفسی بهوی الحبیب فارت
&&
لما رات الکوس دارت
\\
مدت یدها الی رحیق
&&
و النفس بنوره استنارت
\\
لما شربته نفس و ترا
&&
خفت و تصاعدت و طارت
\\
لاقت قمرا اذا تجلی
&&
الشمس من الحیا توارت
\\
جادت بالروح حین لاقت
&&
لا التفتت و لا استشارت
\\
\end{longtable}
\end{center}
