\begin{center}
\section*{غزل شماره ۳۲۱۴: یا مالک دمة الزمان}
\label{sec:3214}
\addcontentsline{toc}{section}{\nameref{sec:3214}}
\begin{longtable}{l p{0.5cm} r}
یا مالک دمة الزمان
&&
یا فاتح جنة الامعانی
\\
لا هوتک موضح المصادر
&&
ناسوتک سلم الامانی
\\
من رام لقاک فی جهات
&&
ردوه بفول لن ترانی
\\
کم اتلفنی بلن حبیبی
&&
لما اتلفنی بلن اتانی
\\
کم رد علی بات وصل
&&
کم عنه رجعت قد دعانی
\\
کم عانق روحه و روحی
&&
کم جالسنی بلا مکان
\\
کم البسنی ببرد تیه
&&
کم اطعمنی و کم سقانی
\\
کم اسکرنی بکاس حب
&&
بین الحرفاء و المغانی
\\
یا قلب کفاک لا تطول
&&
بالله علیک یا لسانی
\\
\end{longtable}
\end{center}
