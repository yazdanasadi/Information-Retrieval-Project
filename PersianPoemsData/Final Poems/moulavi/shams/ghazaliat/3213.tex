\begin{center}
\section*{غزل شماره ۳۲۱۳: املا قدح البقا ندیمی}
\label{sec:3213}
\addcontentsline{toc}{section}{\nameref{sec:3213}}
\begin{longtable}{l p{0.5cm} r}
املا قدح البقا ندیمی!
&&
من خمرة دنک القدیم
\\
صحیح المی و داو سقمی
&&
من غمزة لحظک السقیم
\\
للعشق ظعنت یا مقیما
&&
والظاعن طالب المقیم
\\
قد قیل بمن یراک یوما
&&
بشراک بغیهة النعیم
\\
لایدرک عادلی بعقل
&&
فوارة عشقی القدیم
\\
قدامک روضةالمعالی
&&
ایاک سعاد! ان تقیمی
\\
هل اغد سعاد ذات یوم
&&
سکران بذلک الحریم
\\
تبریز و شمس و دین مولی
&&
ذوالبهجة والید الکریم
\\
\end{longtable}
\end{center}
